\section{Functions}

\subsection*{Student}
\begin{itemize}
	\item Add and drop classes
	\item View attendance
	\item View grade
\end{itemize}

\subsection*{Teacher}
\begin{itemize}
	\item Create a course
	\item Give grade to the students
	\item Start a class, end the class
	\item View student attendance for a certain class
\end{itemize}
